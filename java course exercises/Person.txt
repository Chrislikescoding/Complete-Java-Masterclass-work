        }

            return this.firstName;
        }

            this.lastName = lastName;
        }

            return this.lastName;
        }

            if (age >=0 && age<=100)
                this.age = age;
            else
                this.age = 0;

        }

            return this.age;
        }

            if (this.age >12 && this.age<20) {
