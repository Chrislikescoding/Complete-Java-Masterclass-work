public class AreaCalculator {
    
    public static double area ( double radius) {
    double area = radius * radius * Math.PI;
    if (radius < 0)
        return -1;

         return area;
    }
    public static double area ( double x, double y) {

        if (x < 0  || y < 0) {
            return -1;
        }
        else {
        double area = x * y;

        return area;
        }
    }
     
}
